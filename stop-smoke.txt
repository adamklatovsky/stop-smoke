    <script>
        /**
         * Aplikace Nov√Ω Dech - Logika
         */
        const app = {
            data: {
                startDate: null,
                cigsPerDay: 0,
                costPerPack: 0,
                packSize: 20,
                achievementsUnlocked: []
            },
            
            // Definice √∫spƒõch≈Ø (Gamifikace)

                { id: 'time_24h', title: 'Prvn√≠ den', desc: '24 hodin bez cigarety', thresholdHours: 24, icon: 'üåÖ' },
                { id: 'time_3d', title: 'V√≠kendovk√°≈ô', desc: '3 dny ƒçist√Ω/√°', thresholdHours: 72, icon: 'üßó' },
                { id: 'time_1w', title: 'T√Ωden hrdosti', desc: '1 t√Ωden svobody', thresholdHours: 168, icon: 'üóìÔ∏è' },
                { id: 'time_1m', title: 'Nov√Ω zvyk', desc: '1 mƒõs√≠c bez kou≈ôen√≠', thresholdHours: 720, icon: 'üëë' },
                { id: 'money_500', title: 'Spo≈ôiv√Ω', desc: 'U≈°et≈ôeno 500 Kƒç', thresholdMoney: 500, icon: 'üê∑' },
                { id: 'cigs_100', title: 'ƒåist√© pl√≠ce', desc: '100 nevykou≈ôen√Ωch cigaret', thresholdCigs: 100, icon: 'üå¨Ô∏è' }

                { minutes: 20, title: 'Krevn√≠ tlak', desc: 'Tep a tlak se vracej√≠ k norm√°lu.' },

                { minutes: 1440, title: 'Srdce bez rizika', desc: 'Hladina CO je v norm√°lu. Riziko infarktu kles√°.' }, // 24h
                { minutes: 2880, title: 'Chu≈• a ƒçich', desc: 'Nervov√° zakonƒçen√≠ regeneruj√≠. J√≠dlo chutn√° l√©pe.' }, // 48h
                { minutes: 4320, title: 'D√Ωch√°n√≠', desc: 'Pr≈Ødu≈°ky se uvol≈àuj√≠, energie roste.' } // 72h
            ],

            motivationalQuotes: [
                "Nejlep≈°√≠ ƒças zasadit strom byl p≈ôed 20 lety. Druh√Ω nejlep≈°√≠ ƒças je teƒè.",
                "P≈ôek√°≈æky jsou to, co vid√≠≈°, kdy≈æ p≈ôestane≈° hledƒõt na c√≠l.",
                "Bolest je doƒçasn√°. Hrdost je vƒõƒçn√°.",
                "Tv√© tƒõlo ti dƒõkuje ka≈ædou minutou.",
                "Nen√≠ to o tom, co ztr√°c√≠≈°, ale co z√≠sk√°v√°≈°."
            ],

            init: function() {
                this.loadData();
                
                if (!this.data.startDate) {
                    document.getElementById('setup-screen').classList.remove('hidden');
                    // Nastavit defaultn√≠ ƒças na "teƒè" pro input
                    const now = new Date();
                    now.setMinutes(now.getMinutes() - now.getTimezoneOffset());
                    document.getElementById('startDateInput').value = now.toISOString().slice(0,16);
                } else {
                    this.startDashboard();
                }
            },

            initProfile: function() {
                const startInput = document.getElementById('startDateInput').value;
                const cpd = document.getElementById('cigsPerDay').value;
                const cost = document.getElementById('costPerPack').value;
                const pack = document.getElementById('cigsInPack').value;

                if (!startInput || !cpd || !cost) {
                    alert("Pros√≠m vypl≈à v≈°echna pole.");
                    return;
                }

                this.data.startDate = new Date(startInput).getTime();
                this.data.cigsPerDay = parseInt(cpd);
                this.data.costPerPack = parseInt(cost);
                this.data.packSize = parseInt(pack);
                
                this.saveData();
                document.getElementById('setup-screen').classList.add('hidden');
                this.startDashboard();
            },

            startDashboard: function() {
                document.getElementById('dashboard').classList.remove('hidden');
                this.renderAchievementsLayout();
                this.renderHealthLayout();
                
                // Random cit√°t p≈ôi startu
                const quote = this.motivationalQuotes[Math.floor(Math.random() * this.motivationalQuotes.length)];
                document.getElementById('motivational-quote').innerText = `"${quote}"`;

                // Update loop
                this.update(); 
                setInterval(() => this.update(), 1000);
            },

            update: function() {
                const now = Date.now();
                const diff = now - this.data.startDate;

                if (diff < 0) return; // Datum v budoucnosti

                // ƒåas
                const days = Math.floor(diff / (1000 * 60 * 60 * 24));
                const hours = Math.floor((diff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
                const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
                const totalHours = Math.floor(diff / (1000 * 60 * 60));
                const totalMinutes = Math.floor(diff / (1000 * 60));

                document.getElementById('days').innerText = days;
                document.getElementById('hours').innerText = hours.toString().padStart(2, '0');
                document.getElementById('minutes').innerText = minutes.toString().padStart(2, '0');

                // Statistiky
                const cigsAvoided = (this.data.cigsPerDay / 24 / 60 * totalMinutes).toFixed(1);
                const moneySaved = (cigsAvoided * (this.data.costPerPack / this.data.packSize)).toFixed(0);

                document.getElementById('cigs-not-smoked').innerText = Math.floor(cigsAvoided);
                document.getElementById('money-saved').innerText = `${moneySaved} Kƒç`;

                // Kontrola Achievement≈Ø
                this.checkAchievements(totalHours, parseInt(moneySaved), parseInt(cigsAvoided));
                
                // Kontrola Zdrav√≠
                this.updateHealthTimeline(totalMinutes);
            },

            checkAchievements: function(hours, money, cigs) {
                let newUnlock = false;
                
                this.achievements.forEach(ach => {
                    if (this.data.achievementsUnlocked.includes(ach.id)) return;

                    let unlocked = false;
                    if (ach.thresholdHours && hours >= ach.thresholdHours) unlocked = true;
                    if (ach.thresholdMoney && money >= ach.thresholdMoney) unlocked = true;
                    if (ach.thresholdCigs && cigs >= ach.thresholdCigs) unlocked = true;

                    if (unlocked) {
                        this.data.achievementsUnlocked.push(ach.id);
                        this.showAchievementModal(ach);
                        this.renderAchievementsLayout(); // P≈ôekreslit pro vizu√°ln√≠ efekt
                        newUnlock = true;
                    }
                });

                if (newUnlock) this.saveData();
            },

            renderAchievementsLayout: function() {
                const container = document.getElementById('achievements-list');
                container.innerHTML = '';
                
                this.achievements.forEach(ach => {
                    const isUnlocked = this.data.achievementsUnlocked.includes(ach.id);
                    const el = document.createElement('div');
                    el.className = `badge ${isUnlocked ? 'unlocked' : ''}`;
                    el.title = ach.desc;
                    el.innerHTML = `
                        <i>${isUnlocked ? ach.icon : 'üîí'}</i>
                        <span>${ach.title}</span>
                    `;
                    container.appendChild(el);
                });
            },

            renderHealthLayout: function() {
                const container = document.getElementById('health-timeline');
                container.innerHTML = '';
                this.healthMilestones.forEach((milestone, index) => {
                    const el = document.createElement('li');
                    el.className = 'health-item';
                    el.id = `health-${index}`;
                    el.innerHTML = `
                        <div class="health-progress">
                            <div class="health-title">${milestone.title}</div>
                            <div class="health-desc">${milestone.desc}</div>
                        </div>
                        <div>${index === 0 ? '‚ù§Ô∏è' : '‚öïÔ∏è'}</div>
                    `;
                    container.appendChild(el);
                });
            },

            updateHealthTimeline: function(totalMinutes) {
                this.healthMilestones.forEach((milestone, index) => {
                    const el = document.getElementById(`health-${index}`);
                    if (totalMinutes >= milestone.minutes) {
                        if (!el.classList.contains('unlocked')) {
                            el.classList.add('unlocked');
                        }
                    }
                });
            },

            showAchievementModal: function(ach) {
                document.getElementById('modal-achievement-name').innerText = ach.title;
                document.getElementById('modal-achievement-desc').innerText = ach.desc;
                document.getElementById('achievement-modal').classList.remove('hidden');
            },

            showRelapseModal: function() {
                document.getElementById('relapse-modal').classList.remove('hidden');
            },

            hideRelapseModal: function() {
                document.getElementById('relapse-modal').classList.add('hidden');
            },

            confirmRelapse: function() {
                // Resetovat datum na teƒè, ale nechat nastaven√≠ profilu
                this.data.startDate = Date.now();
                this.saveData();
                this.hideRelapseModal();
                this.update();
                // Volitelnƒõ: Vyƒçistit achievementy? 
                // Z hlediska pozitivn√≠ motivace je lep≈°√≠ nechat star√© achievementy odemƒçen√© 
                // jako p≈ôipom√≠nku, ≈æe to dok√°≈æou, nebo je resetovat. 
                // Zde vol√≠me "mƒõkk√Ω reset" (statistika bƒõ≈æ√≠ od nuly, achievementy z≈Øst√°vaj√≠ jako "historick√© trofeje", nebo je lze smazat pro hardcore m√≥d).
                // Pro tento √∫kol nech√°v√°me achievementy (podpora i p≈ôi ne√∫spƒõchu).
                alert("Nov√Ω zaƒç√°tek. Tentokr√°t to vyjde!");
            },

            saveData: function() {
                localStorage.setItem('novyDechData', JSON.stringify(this.data));
            },

            loadData: function() {
                const saved = localStorage.getItem('novyDechData');
                if (saved) {
                    this.data = JSON.parse(saved);
                }
            }
        };

        // Spustit aplikaci
        app.init();
